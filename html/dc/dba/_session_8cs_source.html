451 Trial GetFirstTrial()
452 {
453 Block firstBlock;
454 try
455 {
456 firstBlock = blocks[0];
457 }
458 catch (ArgumentOutOfRangeException)
459 {
460 throw new NoSuchTrialException("There is no first trial because no blocks have been created!");
461 }
462
463 try
464 {

477 Trial GetLastTrial()
478 {
479 if (blocks.Count == 0) throw new NoSuchTrialException("There is no last trial because no blocks have been created!");
480
481 Block lastValidBlock;
482 int i = blocks.Count - 1;
483 while (i >= 0)
484 {
485 lastValidBlock = blocks[i];
486 if (lastValidBlock.trials.Count > 0)
487 {
488 return lastValidBlock.trials[lastValidBlock.trials.Count - 1];
489 }
490 i--;
491 }
492
493 throw new NoSuchTrialException("There is no last trial, blocks are present but they are all empty.");
494 }

615 public void End()
616 {
617 if (hasInitialised)
618 {
619 isEnding = true;
620 if (InTrial)
621 {
622 try { CurrentTrial.End(); }
623 catch (Exception e) { Debug.LogException(e); }
624 }
625
626 SaveResults();
627
628 try { preSessionEnd.Invoke(this); }
629 catch (Exception e) { Debug.LogException(e); }
630
631 if (storeSessionSettings)
632 {
633 // copy Settings to session folder
634 SaveJSONSerializableObject(new Dictionary<string, object>(settings.baseDict), "settings", dataType: UXFDataType.Settings);
635 }
636
637 if (storeParticipantDetails)
638 {
639 // copy participant details to session folder
640 // we convert to a DataTable because we know the dictionary will be "flat" (one value per key)
641
642 UXFDataTable ppDetailsTable = new UXFDataTable(participantDetails.Keys.ToArray());
643 var row = new UXFDataRow();
644 foreach (var kvp in participantDetails) row.Add((kvp.Key, kvp.Value));
645 ppDetailsTable.AddCompleteRow(row);
646 var ppDetailsLines = ppDetailsTable.GetCSVLines();
647
648 SaveDataTable(ppDetailsTable, "participant_details", dataType: UXFDataType.ParticipantDetails);
649 }
650
651 // end DataHandlers - forces completion of tasks
652 foreach (var dataHandler in ActiveDataHandlers)
653 {
654 try { dataHandler.CleanUp(); }
655 catch (Exception e) { Debug.LogException(e); }
656 }
657
658 try { onSessionEnd.Invoke(this); }
659 catch (Exception e) { Debug.LogException(e); }
660
661 currentTrialNum = 0;
662 currentBlockNum = 0;
663 blocks = new List<Block>();
664 _hasInitialised = false;
665
666 Utilities.UXFDebugLog("Ended session.");
667 isEnding = false;
668 }

671 void SaveResults()
672 {
673 // generate list of all headers possible
674 // hashset keeps unique set of keys
675 HashSet<string> resultsHeaders = new HashSet<string>();
676 foreach (Trial t in Trials)
677 if (t.result != null)
678 foreach (string key in t.result.Keys)
679 resultsHeaders.Add(key);
680
681 UXFDataTable table = new UXFDataTable(Trials.Count(), resultsHeaders.ToArray());
682 foreach (Trial t in Trials)
683 {
684 if (t.result != null)
685 {
686 UXFDataRow row = new UXFDataRow();
687 foreach (string h in resultsHeaders)
688 {
689 if (t.result.ContainsKey(h) && t.result[h] != null)
690 {
691 row.Add(( h, t.result[h] ));
692 }
693 else
694 {
695 row.Add(( h, string.Empty ));
696 }
697 }
698 table.AddCompleteRow(row);
699 }
700 }
701
702 SaveDataTable(table, "trial_results", dataType: UXFDataType.TrialResults);
703 }
