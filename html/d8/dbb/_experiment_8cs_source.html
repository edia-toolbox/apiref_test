98 bool IsValid() {
99 bool isValid = true;
100 List<string> msgs = new();
101
102 // Are there executers?
103 if (XBlockExecuters == null || XBlockExecuters.Count == 0) {
104 isValid = false;
105 msgs.Add("XBLock Executers list is empty!");
106 }
107
108 // Are there executers with the same name?
109 var names = XBlockExecuters.Select(g => g.name);
110 if (names.Count() != names.Distinct().Count()) {
111 msgs.Add("All XBlock Executers need unique names!");
112 isValid = false;
113 }
114
115 // Are the gameobjects in Experiment.blocks properly named? <type>_<subtype>
116 foreach (XBlock g in XBlockExecuters) {
117 if (!Regex.IsMatch(g.name, @"^[a-z0-9]+-[a-z0-9\-'().,_]+$")) {
118 msgs.Add($"Invalid gameobject (XBlock Executer) naming format found in: <b>{g.name}</b>; must adhere to: <type>-<subtype>");
119 isValid = false;
120 }
121 }
122
123 if (!isValid) {
124 foreach(string s in msgs)
125 Debug.LogErrorFormat(s);
126
127 // TODO: Think of a way to alert the user in a build
128 //ShowMessageToExperimenter(msg, false);
129 }
130
131 return isValid;
132 }

412 void BlockStart() {
413 AddToLog("Block Start");
414
415 // Set new storedBlockNum value
416 _activeSessionBlockNum = Session.instance.currentBlockNum;
417 _activeXBlock = XBlockExecuters[XBlockExecuters.FindIndex(x => x.name == Session.instance.CurrentBlock.settings.GetString("_assetId"))];
418 _activeXBlock.enabled = true;
419 _activeXBlock.gameObject.SetActive(true);
420 _activeXBlock.OnBlockStart();
421
422 // Update block progress
423 UpdateBlockProgress();
424
425 // Check for block introduction flag
426 bool hasIntro = Session.instance.CurrentBlock.settings.GetStringList("_start").Count > 0;
427
428 // Inject introduction step or continue UXF sequence
429 if (hasIntro) {
430 EventManager.StartListening(Edia.Events.StateMachine.EvProceed, BlockContinueAfterIntro); // listener as it event call can come from any script
431 ShowMessageToUser (Session.instance.CurrentBlock.settings.GetStringList("_start"));
432 UpdateProgressStatus(Session.instance.CurrentBlock.settings.GetString("blockId") + " Introduction");
433 }
434 else {
435 StartTrial();
436 UpdateProgressStatus(Session.instance.CurrentBlock.settings.GetString("blockId"));
437 }
438 }
439
440 void BlockEnd() {
441 _activeXBlock.OnBlockEnd();
442
443 // Check for block outro flag
444 bool hasOutro = Session.instance.CurrentBlock.settings.GetStringList("_end").Count > 0;
445
446 if (hasOutro) {
447 EventManager.StartListening(Edia.Events.StateMachine.EvProceed, BlockContinueAfterOutro); // listener as it event call can come from any script
448 EnableProceedButton(true);
449 ShowMessageToUser(Session.instance.CurrentBlock.settings.GetStringList("_end"));
450 UpdateProgressStatus("Block Outro");
451 _activeXBlock.OnBlockOutro();
452 }
453 else {
454 BlockCheckAndContinue();
455 }
456 }

459

505 void OnTrialBeginUXF(Trial newTrial) {
506 AddToExecutionOrderLog("OnTrialBeginUXF");
507
508 XRManager.Instance.ShowVRInstantly(); // TODO: This is now assuming that we want VR visable at the start of each trial.

520
522 void OnTrialEndUXF(Trial endedTrial) {
523 AddToExecutionOrderLog("OnTrialEnd");
524 SaveCustomDataTables();
525
526 _activeXBlock.OnEndTrial();
527
528 // Are we ending?
529 if (Session.instance.isEnding)
530 return;
531
532 // Is there a PAUSE requested right now?
533 if (_isPauseRequested) {
534 _isPauseRequested = false;
535
536 if (endedTrial == Session.instance.LastTrial)
537 return;
538
539 AddToExecutionOrderLog("Injected SessionBreak");
540 SessionPause();
541 return;
542 }
543
544 // Reached last trial in a block?
545 if (Session.instance.CurrentBlock.lastTrial != endedTrial) { // NO
546 Session.instance.BeginNextTrialSafe();
547 return;
548 }
549 else {
550 BlockEnd(); // YES
551 return;
552 }

595
596 if (_proceedTimer != null) {
597 StopCoroutine(_proceedTimer); // Kill timer, if any

607 InBetweenSteps();
608 UpdateStepProgress();
609 _activeXBlock.trialSteps[_currentStep].Invoke();
610
611 }
612 else EndTrial();
613 }

658 public void ConvertAndSaveDataToUXF(string[] headers, List<string[]> values, string filename) {
659 var UXFheaders = headers;

661

665 newRow.Add((UXFheaders[s], valuerow[s]));
