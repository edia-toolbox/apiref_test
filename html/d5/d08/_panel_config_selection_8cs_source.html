106 void LoadJsons() {
107
108 // Session info
109 string currentPath = Constants.PathToParticipantFiles + _subject + "/" + _session + "/";
110 _sessionInfoJsonString = FileManager.ReadStringFromApplicationPath(currentPath + Constants.FileNameSessionInfo);
111
112 // Block sequence
113 string eBSequenceFilePath = currentPath + Constants.FileNameSessionSequence;
114 _xBlockSequenceJsonString = FileManager.ReadStringFromApplicationPath(eBSequenceFilePath);
115
116 // Task definitions
117 string[] filelist = FileManager.GetAllFilenamesWithExtensionFrom(Constants.PathToBaseDefinitions, "json");
118
119 foreach (string s in filelist) {
120 _taskDefinitionJsonStrings.Add(FileManager.ReadStringFromApplicationPath(Constants.PathToBaseDefinitions+ s));
121 }
122
123 // Block Task Definitions
124 xBlockDefinitionsFileList = FileManager.GetAllFilenamesWithExtensionFrom(currentPath + Constants.FolderNameXBlockDefinitions, "json").ToList();
125
126 foreach (string s in xBlockDefinitionsFileList) {
127 string currentFileName = currentPath + "/" + Constants.FolderNameXBlockDefinitions + "/" + s;
128 _xBlockDefinitionJsonStrings.Add(FileManager.ReadStringFromApplicationPath(currentFileName.ToLower()));
129 }

132 private bool ValidateConfigs() {
133 // Validate all strings on formatting
134
135 if (!ValidateJSON(_sessionInfoJsonString)) {
136 Debug.Log("JSON Format error: Session Info");
137 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: session_info.json", false ));
138 return false;
139 }
140
141 if (!ValidateJSON(_xBlockSequenceJsonString)) {
142 Debug.Log("JSON Format error: Xblock sequence");
143 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: session_sequence.json", false));
144 return false;
145 }
146
147 foreach (string s in _taskDefinitionJsonStrings) {
148 if (!ValidateJSON(s)) {
149 Debug.Log("JSON Format error: Task definition");
150 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: Task definition", false));
151 return false;
152 }
153 }
154
155 foreach (string s in _xBlockDefinitionJsonStrings) {
156 if (!ValidateJSON(s)) {
157 Debug.Log("JSON Format error: Xblock definition");
158 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: Xblock definition", false));
159 return false;
160 }
161 }
162
163 // Sanity check - for each entry in the sequence file, there should be a file with the same name
164 _xBlockSequence = UnityEngine.JsonUtility.FromJson<XBlockSequence>(_xBlockSequenceJsonString);
165
166 foreach (string s in _xBlockSequence.Sequence) {
167 bool found = false;
168 //Debug.Log(" src: " + s.ToLower() );
169 foreach ( string deffile in xBlockDefinitionsFileList ) {
170 //Debug.Log(" check: " + deffile.Split('.')[0].ToLower());
171 if ( s.ToLower() == deffile.Split('.')[0].ToLower() ) {
172 found = true;
173 break;
174 }
175 }
176 if (!found) {
177 Debug.LogError("No <b>" + s + ".json</b> config file not found in xblock definitions folder");
178 return false;
179 }
180 }
181
182 return true;
183 }
