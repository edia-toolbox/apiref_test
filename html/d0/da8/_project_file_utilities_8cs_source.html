29 public static void UpdateChangelog () {
30
31 logSRC = new List<ChangeLogEntry> ();
32 logOrderedByVersion = new List<ChangeLogEntry> ();
33
34 string CSVlog = FileManager.ReadString ("ChangeLog.csv");
35 string[] rows = CSVlog.Split ('\n');
36
37 foreach (string s in rows) {
38 string[] values = s.Split (',');
39
41 ChangeLogEntry newEntry = new ChangeLogEntry ();
42 newEntry.description = values[0];
43 newEntry.type = values[1];
44 newEntry.version = values[2];
45 newEntry.date = values[3];
46 newEntry.dev = values[4];
47
48 logSRC.Add (newEntry);
49 }
50
51 // Remove the first one, as those are the headers
52 logSRC.RemoveAt (0);
53
54 // Sort on versions
55 logOrderedByVersion = logSRC.OrderByDescending (c => c.version).ToList ();
56
57 // CHANGELOG header
58 changeLogString = string.Empty;
59 changeLogString += "# Changelog\n\nAll notable changes to this project will be documented in this file.\nThe format is based on [Keep a Changelog](https://keepachangelog.com/en/1.0.0/),\nand this project adheres to [Semantic Versioning](https://semver.org/spec/v2.0.0.html).\n\n";
60
61 // Add sections per version
62 while (logOrderedByVersion.Count > 0)
63 GenerateSingleVersionEntry ();
64
65 // Write result to file
66 FileManager.WriteString ("/../CHANGELOG.md", changeLogString, true);
67
68 Debug.Log ("Generated new CHANGELOG.md");
69 }

72 public static void GenerateSingleVersionEntry () {
73
74 if (logOrderedByVersion.Count == 0)
75 return;
76
77 logSingleVersion = new List<ChangeLogEntry> ();
78 string versioncheck = logOrderedByVersion[0].version;
79 bool CheckForVersion = true;
80
81 while (CheckForVersion) {
82
83 foreach (ChangeLogEntry c in logOrderedByVersion) {
84 if (c.version == versioncheck) {
85 logSingleVersion.Add (c);
86 logOrderedByVersion.Remove (c);
87 break;
88 }
89
90 CheckForVersion = false;
91 }
92
93 if (logOrderedByVersion.Count == 0 || !CheckForVersion) // array depleted or last was found, stop while loop.
94 break;
95 }
96
97 // Sort the list on type and dates
98 logOrderedOnType = new List<ChangeLogEntry> ();
99 logOrderedOnType = logSingleVersion.OrderBy (c => c.type).ThenByDescending (c => c.date).ToList ();
100
101 // Add ChangeLog section for this version
102 changeLogString += "\n## [" + logOrderedOnType[0].version + "] - " + logOrderedOnType[0].date;
103
104 string currentType = string.Empty;
105
106 foreach (ChangeLogEntry c in logOrderedOnType) {
107 if (c.type != currentType) { // new changetype header
108 changeLogString += "\n### " + c.type + "\n";
109 currentType = c.type;
110 }
111
112 changeLogString += "- " + c.description + "\n";
113 }
114
115 logOrderedOnType.Clear();
116 }
117 }
