118 public static void CopyAll(DirectoryInfo source, DirectoryInfo target, string exclude)
119 {
120 Directory.CreateDirectory(target.FullName);
121
122 // Copy each file into the new directory.
123 foreach (FileInfo fi in source.GetFiles())
124 {
125 // Debug.Log((@"Copying {0}\{1}", target.FullName, fi.Name));
126 if (exclude.Length == 0 || (!fi.Name.Contains(exclude)))
127 fi.CopyTo(Path.Combine(target.FullName, fi.Name), true);
128 }
129
130 // Copy each subdirectory using recursion.
131 foreach (DirectoryInfo diSourceSubDir in source.GetDirectories())
132 {
133 DirectoryInfo nextTargetSubDir =
134 target.CreateSubdirectory(diSourceSubDir.Name);
135 CopyAll(diSourceSubDir, nextTargetSubDir,exclude);
136 }
137 }
