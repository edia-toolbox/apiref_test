79 public void Begin()
80 {
81 if (!session.hasInitialised)
82 {
83 throw new InvalidOperationException("Cannot begin trial, session is is not ready! Session has not been started yet with session.Begin() (or via the UI), or session has already ended.");
84 }
85 if (session.InTrial) session.CurrentTrial.End();
86
87 session.currentTrialNum = number;
88 session.currentBlockNum = block.number;
89
90 status = TrialStatus.InProgress;
91 startTime = Time.time;
92 result = new ResultsDictionary(session.Headers, true);
93
94 result["experiment"] = session.experimentName;
95 result["ppid"] = session.ppid;
96 result["session_num"] = session.number;
97 result["trial_num"] = number;
98 result["block_num"] = block.number;
99 result["trial_num_in_block"] = numberInBlock;
100 result["start_time"] = startTime;
101
102 foreach (Tracker tracker in session.trackedObjects)
103 {
104 try
105 {
106 tracker.StartRecording();
107 }
108 catch (NullReferenceException)
109 {
110 Utilities.UXFDebugLogWarning("An item in the Tracked Objects field of the UXF session if empty (null)!");
111 }
112 }
113 session.onTrialBegin.Invoke(this);
114 }
115
119 public void End()
120 {
121 status = TrialStatus.Done;
122 endTime = Time.time;
123 result["end_time"] = endTime;
124
125 // check no duplicate trackers
126 List<string> duplicateTrackers = session.trackedObjects.Where(tracker => tracker != null)
127 .GroupBy(tracker => tracker.dataName)
128 .Where(g => g.Count() > 1)
129 .Select(y => y.Key)
130 .ToList();
131
132 if (duplicateTrackers.Any()) throw new InvalidOperationException(string.Format("Two or more trackers in the Tracked Objects field in the Session Inspector have the following object name and descriptor pair, please change the object name fields on the trackers to make them unique: {0}", string.Join(",", duplicateTrackers)));
133
134 // log tracked objects
135 foreach (Tracker tracker in session.trackedObjects)
136 {
137 try
138 {
139 tracker.StopRecording();
140 SaveDataTable(tracker.data, tracker.dataName, dataType: UXFDataType.Trackers);
141 }
142 catch (NullReferenceException)
143 {
144 Utilities.UXFDebugLogWarning("An item in the Tracked Objects field of the UXF session if empty (null)!");
145 }
146 }
147
148 // log any settings we need to for this trial
149 foreach (string s in session.settingsToLog)
150 {
151 result[s] = settings.GetObject(s, string.Empty);
152 }
153
154 session.onTrialEnd.Invoke(this);
155 }

157 public bool CheckDataTypeIsValid(string dataName, UXFDataType dataType)
158 {
159 if (dataType.GetDataLevel() != UXFDataLevel.PerTrial)
160 {
161 Utilities.UXFDebugLogErrorFormat(
162 "Error trying to save data '{0}' of type UXFDataType.{1} associated with the Trial. The valid types for this method are {2}. Reverting to type UXFDataType.OtherTrialData.",
163 dataName,
164 dataType,
165 string.Join(", ", UXFDataLevel.PerTrial.GetValidDataTypes())
166 );
167 return false;
168 }
169 else
170 {
171 return true;
172 }

181 public void SaveDataTable(UXFDataTable table, string dataName, UXFDataType dataType = UXFDataType.OtherTrialData)
182 {
183 if (!CheckDataTypeIsValid(dataName, dataType)) dataType = UXFDataType.OtherTrialData;
184
185 int i = 0;

189 result[string.Format("{0}_location_{1}", dataName, i++)] = location.Replace("\\", "/");
190 }
