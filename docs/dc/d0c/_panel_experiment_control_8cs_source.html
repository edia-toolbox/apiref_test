146 void OnEvEnableButton(eParam e) {
147
148 bool turnOn = e.GetStrings()[1].ToUpper() == "TRUE";
149
150 //Debug.Log("OnEvEnableButton " + e.GetStrings()[0].ToUpper() + ":" + turnOn);
151
152 switch (e.GetStrings()[0].ToUpper()) {
153 case "PAUSE":
154 SetButtonState(btnPauseExperiment, turnOn);
155 break;
156 case "PROCEED":
157 SetButtonState(btnProceedExperiment, turnOn);
158 StartCoroutine("ChangeColorOverTime", btnProceedExperiment.GetComponent<Image>());
159 break;
160 }
161 }
162
163 IEnumerator ChangeColorOverTime(Image image) {
164 Color endColor = Color.white;
165 Color flashColor = new Color(0.2745098f, 0.6352941f, 0.8000001f);
166 float duration = 0.3f;
167 float elapsedTime = 0f;
168
169 while (elapsedTime < duration) {
170 image.color = Color.Lerp(flashColor, endColor, elapsedTime / duration);
171 elapsedTime += Time.deltaTime;
172 yield return null; // Wait for the next frame
173 }
174
175 // Ensure the final color is set
176 image.color = endColor;
177 }
