180 private void LoadNewScene(int sceneIndex) {
181 if (!isLoading)
182 StartCoroutine(LoadScene(sceneIndex));

184
185 private IEnumerator LoadScene(int sceneIndex) {
186 this.Add2Console("Loading scene: " + scenes[sceneIndex].name);
187 OpenMenu(false);
188
189 isLoading = true;
190
191 OnLoadBegin?.Invoke();
192 yield return screenFader.StartFadeBlackIn();
193
194 yield return new WaitForSeconds(0.5f);
195
196 yield return StartCoroutine(LoadNew(sceneIndex));
197 yield return screenFader.StartFadeBlackOut();
198
199 OnLoadEnd?.Invoke();
200
201 isLoading = false;
202 currentSceneIndex = sceneIndex;
203
204 yield return null;
205 }
206
207 private IEnumerator LoadNew(int sceneIndex) {
208
209 AsyncOperation loadOperation = SceneManager.LoadSceneAsync(scenes[sceneIndex].name, LoadSceneMode.Single);
210
211 while (!loadOperation.isDone)
212 yield return null;
213 }
