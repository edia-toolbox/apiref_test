114 protected override Vector3 ComputeDesiredMove(Vector2 input)
115 {
116 // Don't need to do anything if the total input is zero.
117 // This is the same check as the base method.
118 if (input == Vector2.zero)
119 return Vector3.zero;
120
121 // Initialize the Head Transform if necessary, getting the Camera from XR Origin
122 if (m_HeadTransform == null)
123 {
124 var xrOrigin = system.xrOrigin;
125 if (xrOrigin != null)
126 {
127 var xrCamera = xrOrigin.Camera;
128 if (xrCamera != null)
129 m_HeadTransform = xrCamera.transform;
130 }
131 }
132
133 // Get the forward source for the left hand input
134 switch (m_LeftHandMovementDirection)
135 {
136 case MovementDirection.HeadRelative:
137 if (m_HeadTransform != null)
138 m_LeftMovementPose = m_HeadTransform.GetWorldPose();
139
140 break;
141
142 case MovementDirection.HandRelative:
143 if (m_LeftControllerTransform != null)
144 m_LeftMovementPose = m_LeftControllerTransform.GetWorldPose();
145
146 break;
147
148 default:
149 Assert.IsTrue(false, $"Unhandled {nameof(MovementDirection)}={m_LeftHandMovementDirection}");
150 break;
151 }
152
153 // Get the forward source for the right hand input
154 switch (m_RightHandMovementDirection)
155 {
156 case MovementDirection.HeadRelative:
157 if (m_HeadTransform != null)
158 m_RightMovementPose = m_HeadTransform.GetWorldPose();
159
160 break;
161
162 case MovementDirection.HandRelative:
163 if (m_RightControllerTransform != null)
164 m_RightMovementPose = m_RightControllerTransform.GetWorldPose();
165
166 break;
167
168 default:
169 Assert.IsTrue(false, $"Unhandled {nameof(MovementDirection)}={m_RightHandMovementDirection}");
170 break;
171 }
172
173 // Combine the two poses into the forward source based on the magnitude of input
174 var leftHandValue = leftHandMoveAction.action?.ReadValue<Vector2>() ?? Vector2.zero;
175 var rightHandValue = rightHandMoveAction.action?.ReadValue<Vector2>() ?? Vector2.zero;
176
177 var totalSqrMagnitude = leftHandValue.sqrMagnitude + rightHandValue.sqrMagnitude;
178 var leftHandBlend = 0.5f;
179 if (totalSqrMagnitude > Mathf.Epsilon)
180 leftHandBlend = leftHandValue.sqrMagnitude / totalSqrMagnitude;
181
182 var combinedPosition = Vector3.Lerp(m_RightMovementPose.position, m_LeftMovementPose.position, leftHandBlend);
183 var combinedRotation = Quaternion.Slerp(m_RightMovementPose.rotation, m_LeftMovementPose.rotation, leftHandBlend);
184 m_CombinedTransform.SetPositionAndRotation(combinedPosition, combinedRotation);
185
186 return base.ComputeDesiredMove(input);
187 }
