49 public static UXFDataTable FromCSV(string[] csvLines)
50 {
51 string[] headers = csvLines[0].Split(',');
52 var table = new UXFDataTable(csvLines.Length - 1, headers);
53
54 // traverse down rows
55 for (int i = 1; i < csvLines.Length; i++)
56 {
57 string[] values = csvLines[i].Split(',');
58
59 // if last line, just 1 item in the row, and it is blank, then ignore it
60 if (i == csvLines.Length - 1 && values.Length == 1 && values[0].Trim() == string.Empty ) break;
61
62 // check if number of columns is correct
63 if (values.Length != headers.Length) throw new Exception($"CSV line {i} has {values.Length} columns, but expected {headers.Length}");
64
65 // build across the row
66 var row = new UXFDataRow();
67 for (int j = 0; j < values.Length; j++)
68 row.Add((headers[j], values[j]));
69
70 table.AddCompleteRow(row);
71 }
72
73 return table;
74 }

145 static string FormatItem(object item, CultureInfo culture, string decimalFormat = "0.######")
146 {
147 switch (item)
148 {
149 case sbyte sbyteNum: return sbyteNum.ToString(culture);
150 case byte byteNum: return byteNum.ToString(culture);
151 case short shortNum: return shortNum.ToString(culture);
152 case ushort ushortNum: return ushortNum.ToString(culture);
153 case int intNum: return intNum.ToString(culture);
154 case uint uintNum: return uintNum.ToString(culture);
155 case long longNum: return longNum.ToString(culture);
156 case ulong ulongNum: return ulongNum.ToString(culture);
157 case float floatNum: return floatNum.ToString(decimalFormat, culture);
158 case double doubleNum: return doubleNum.ToString(decimalFormat, culture);
159 case decimal decimalNum: return decimalNum.ToString(decimalFormat, culture);
160 case null: return "null";
161 default: return item.ToString().Replace(culture.TextInfo.ListSeparator, "_");
162 };
